(function (angular) {
    "use strict";
 
    angular.module('myApp').service('PlaygroundService', PlaygroundService);
 
    function PlaygroundService() {
        var playground = generateEmptyPlayground();
 
        function generateEmptyPlayground(options) {
            var i, j,
                rowCount = 3,
                colCount = 3,
                result = [];
 
            for(i = 0; i < rowCount; i += 1) {
                result[i] = [];
                for(j = 0; j < colCount; j += 1) {
                    result[i][j] = {
                        x: options && options.fakeData ? !!Math.random().round() : false,
                        o: options && options.fakeData ? !!Math.random().round() : false
                    };
                }
            }
            return result;
        }
 
        return {
            getPlayground: function () {
                return playground;
            },
 
            setX: function (row, col) {
                playground[row][col].x = true;
            },
 
            setO: function (row, col) {
                playground[row][col].o = true;
            },
 
            newGame: function () {
                playground = generateEmptyPlayground();
 
                return result;
            }
        }
    }
 
}(angular))
