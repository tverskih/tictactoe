(function (angular) {
    "use strict";
 
    angular.module('myApp').factory('AIModel', AIModelFactory);
 
    function AIModelFactory() {
        function AIModel(playground) {
            this.playground = playground;
        }
 
        var prototype = AIModel.prototype;
 
        prototype.makeTurnForSymbol = function (playSymbol) {
            var playground = this.playground;
 
            /* Case 1 */
            // if no cells is available
            // or playground already has winning line
            // don't make turn and return false
            if (playground.isFinished()) { return false; }
 
            var lines       = playground.getLines(),
                cells       = playground.getCells(),
                centerCell  = playground.getCenterCell(),
                cornerCells = playground.getCornerCells(),
                turnWasMade = false,
                emptyCell,
                i;
 
            /* Case 2 */
            // if playground is empty then set to center
            if (playground.isEmpty()) {
                centerCell.setSymbol(playSymbol);
 
                return true;
            }
 
            /* Case 3 */
            // if on some line 2 player's symbols OR
            // if on some line 2 enemy's symbols
            // THEN set 3rd
            for(i = 0; i < lines.length; i += 1) {
                emptyCell = lines[i].getEmptyCell();
                if ((
                        lines[i].countSymbol(playSymbol)      === 2 ||
                        lines[i].countEnemySymbol(playSymbol) === 2
                    ) && emptyCell) {
 
                    emptyCell.setSymbol(playSymbol);
 
                    return true;
                }
 
            }
 
            /* Case 4 */
            // on other cases find a free corner
            for (i = 0; i < cornerCells.length; i += 1) {
                if (!cornerCells[i].isSet()) {
                    cornerCells[i].setSymbol(playSymbol);
 
                    return true;
                }
            }
        };
 
        return AIModel;
    }
 
}(angular))
